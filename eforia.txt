
CREATE DATABASE eforia;
USE eforia;

CREATE TABLE Poliths(
AFM char(9) PRIMARY KEY ,
ONOMA_EPONYMO varchar(40) NOT NULL ,
DIEYTHYNSH varchar(30)
);

CREATE TABLE Foros(
KODIKOS char(5) PRIMARY KEY, 
ONOMASIA varchar(30) NOT NULL,
EPIBARYNSH decimal(4,2), 
ETOS int(4)
);

CREATE TABLE Epibarynsh(
KOD_FOROY char(5) ,
AFM_POLITH CHAR(9),
ONOMASIA varchar(30) ,
EPIBARYNSH decimal(5,2), 
ETOS int(4),
PRIMARY KEY (KOD_FOROY, AFM_POLITH),
FOREIGN KEY (AFM_POLITH) REFERENCES Poliths(AFM),
FOREIGN KEY (KOD_FOROY) REFERENCES Foros(KODIKOS)
);

CREATE TABLE Dhlosh(
ARITHOS_DHLOS char(12) PRIMARY KEY ,
ETOS year,
EISODIMA decimal(10,2), 
AFM_POLITH char(9),
FOREIGN KEY (AFM_POLITH) REFERENCES Poliths(AFM)
);
